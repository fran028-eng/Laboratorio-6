#Practica 1 - Cifrado 1pts

#1️-Instalar GPG2

#En Parrot OS normalmente ya viene instalado, pero por si no:

sudo apt update && sudo apt install gnupg2 -y


#2-Generar tus claves GPG

#Si es la primera vez que lo usas:

gpg2 --full-generate-key


#Comprobar que la clave existe:
gpg2 --list-keys



#3️ Crear carpeta y archivo a cifrar

mkdir -p ~/cifrado_gpg
echo "Este es un archivo secreto" > ~/cifrado_gpg/secreto.txt


#4️-Cifrar el archivo

#Busca tu ID de clave:
gpg2 --list-keys

#Cifra el archivo:
gpg2 -e -r "TU_ID_O_EMAIL" ~/cifrado_gpg/secreto.txt

#5️- Intentar abrir el archivo cifrado


cat ~/cifrado_gpg/secreto.txt.gpg
#Verás texto ilegible (datos binarios).

#6-Descifrar el archivo

gpg2 -d ~/cifrado_gpg/secreto.txt.gpg > ~/cifrado_gpg/secreto_descifrado.txt


#7️- Verificar contenido descifrado

cat ~/cifrado_gpg/secreto_descifrado.txt
